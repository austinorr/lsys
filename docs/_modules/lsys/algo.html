[docs]def midpoints(np_array):
    """
    Converts np array to an array of midpoints ignoring

    Parameters
    ----------
    np_array : array-like
        1-dimensional data

    Returns
    -------
    1-D numpy.array
        The length of the output is one less than
        the length of the input array.

    """
    i = 1

    np_array = validate.is_np(np_array)

    result = (np_array[~numpy.isnan(np_array)][i:] +
              np_array[~numpy.isnan(np_array)][:-i]) / 2

    return result

[docs]def xy_to_coords(x, y, stride=None):
    _x = validate.is_np(x)
    _y = validate.is_np(y)

    segments = numpy.array([x, y]).T.reshape(-1, 1, 2)
    segments = numpy.concatenate([segments[:-1], segments[1:]], axis=1)

    if stride is not None:
        return segments[0::stride]

    return segments
